a=int(input("enter the first num "))
o=input("enter the operator (+,-,*,/): ")
b=int(input("enter the second num "))
if o=="+":
    if a==56 and b==9:
        print(77)
    else:
        print(a+b)
if o=="*":
    if a==5 and b==6:
        print(45)
    else:
        print(a*b)
if o=="/":
    if a==56 and b==4:
        print(6)
    else:
        print(a/b)
if o=="-":
    print(a-b)
